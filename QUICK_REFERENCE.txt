  // Paste into browser console for quick benchmark

  async function benchmark() {
    const modes = ['sequential', 'parallel'];
    const results = {};
    
    for (const mode of modes) {
      console.log(`\n=== Running ${mode} mode ===`);
      const start = performance.now();
      
      // Configure test parameters
      const testParams = {
        teamSize: 4,
        popSize: 200,
        maxGen: 500,
        crossRate: 0.85,
        mutRate: 0.10,
        eliteRate: 0.05,
        ensembleN: 20,
        weights: {wSkill: 0.55, wMbti: 0.20, wBal: 0.25}
      };
      
      if (mode === 'sequential') {
        window.DEBUG_FORCE_SEQUENTIAL = true;
        await runGA();
      } else {
        window.DEBUG_FORCE_SEQUENTIAL = false;
        await runGA();
      }
      
      const elapsed = performance.now() - start;
      results[mode] = {
        time: elapsed,
        timeFormatted: (elapsed / 1000).toFixed(2) + 's'
      };
      
      console.log(`${mode} completed in ${results[mode].timeFormatted}`);
    }
    
    const speedup = results.sequential.time / results.parallel.time;
    const cores = navigator.hardwareConcurrency || 4;
    const efficiency = (speedup / (cores - 1)) * 100;
    
    console.log('\n╔═══════════════════════════════════╗');
    console.log('║     BENCHMARK RESULTS             ║');
    console.log('╠═══════════════════════════════════╣');
    console.log(`║ Sequential:  ${results.sequential.timeFormatted.padEnd(18)}║`);
    console.log(`║ Parallel:    ${results.parallel.timeFormatted.padEnd(18)}║`);
    console.log(`║ Speedup:     ${speedup.toFixed(2)}x${' '.repeat(16)}║`);
    console.log(`║ Efficiency:  ${efficiency.toFixed(1)}%${' '.repeat(15)}║`);
    console.log(`║ CPU Cores:   ${cores}${' '.repeat(18)}║`);
    console.log('╚═══════════════════════════════════╝');
    
    return results;
  }

  // Run benchmark
  benchmark();
